Урок 10. Распознавание лиц и эмоций¶

Нужно написать приложение, которое будет получать на вход изображение.

В процессе определять, что перед камерой находится человек, задетектировав его лицо на кадре.

На изображении человек показывает жесты руками, а алгоритм должен считать их и классифицировать.

def protobuf_to_dict(proto_obj):
    key_list = proto_obj.DESCRIPTOR.fields_by_name.keys()
    d = {}
    for key in key_list:
        d[key] = getattr(proto_obj, key)
    return d
    
def process_image(model, file):
    with mp_face_detection.FaceDetection() as face_detection:
        with mp_hands.Hands(max_num_hands=1) as hands:
            # для потока с камеры
            # cam = cv2.VideoCapture(0)
            # while cap.isOpened():
            #     success, image = cam.read()
            #     if not success: # Пустой фрейм с камеры
            #         continue
            
                text = ''
            
            # для одной фотки
                image = cv2.imread(file)
    
            # цикл обработки
                image = cv2.cvtColor(image, cv2.COLOR_BGR2RGB)
                results = face_detection.process(image)
                
                # Порог сраюатывание на "лицо"
                if results.detections and MessageToDict(results.detections[0])['score'][0] > 0.85:
                    text += 'Есть лицо. '
                    
                    results = hands.process(image)
                    
                    if results.multi_hand_landmarks:
                        gesture = model.predict_proba(landmarks_to_df(results.multi_hand_landmarks))[0]
                        
                        if max(gesture) > 0.5:
                            gesture = np.argmax(gesture)
                            
                            match gesture:
                                case 0:
                                    text += 'Жест не определён'
                                case 1:
                                    text += 'Жест "указательный и большой палец"'
                                case 2:
                                    text += 'Жест "OK"'
                                case 3:
                                    text += 'Жест "С"'
                        else:
                            text += 'Нет жестов на изображении. '
                    else:
                        text += 'Нет жестов! '
                
                else:
                    text += 'Нет лица, поэтому жест не определяем'

                plt.figure(figsize=(5, 4), dpi=80)
                plt.axis('off')
                plt.imshow(image)
                plt.title(text)

def landmarks_to_df(landmarks, df=None, target=None):
    """Функция масштабирует точки жестов в координаты, чтобы 
    фигуры определялись в любой точке кадра.
    """
    landmarks = MessageToDict(landmarks[0])['landmark']
    
    # координаты
    x = np.empty(len(landmarks))
    y = np.empty(len(landmarks))
    z = np.empty(len(landmarks))
    
    for i, v in enumerate(landmarks):
        x[i] = v['x']
        y[i] = v['y']
        z[i] = v['z']
    
    # мсшабируем
    scaler = MinMaxScaler()
    x = scaler.fit_transform(x.reshape(-1, 1)).reshape(1, -1)
    y = scaler.fit_transform(y.reshape(-1, 1)).reshape(1, -1)
    z = scaler.fit_transform(z.reshape(-1, 1)).reshape(1, -1)
    
    if target is not None:
        features = np.c_[x, y, z, target]
    else:
        features = np.c_[x, y, z]
    
    if df is not None:
        df = pd.concat((df, pd.DataFrame(features)), axis=0)
    else:
        df = pd.DataFrame(features)
    
    return df

def process_files_chunk(files):
    """Вносим координаты точек в данные картинок, присваиваем классы жестам"""
    mapping_dict = {'02': 1, '07': 2, '09': 3}
    df = pd.DataFrame()
    with mp_hands.Hands(max_num_hands=1, static_image_mode=True) as hands:
        for idx, file in enumerate(files):
            image = cv2.imread(file)
            results = hands.process(cv2.cvtColor(image, cv2.COLOR_BGR2RGB))
        
            if not results.multi_hand_landmarks:
                continue
            
            target = mapping_dict.get(re.findall('frame_\d{2}_(\d{2})', file)[0], 0)
            df = landmarks_to_df(results.multi_hand_landmarks, df, target)

    return df
